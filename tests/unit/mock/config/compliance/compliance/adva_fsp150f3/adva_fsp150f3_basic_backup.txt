    configure access-port access-1-1-1-3
      add flow flow-1-1-1-3-1 "WTRL" regular-evc disabled disabled disabled disabled 0 disabled none none "1138-*" 100160000 0 access-interface access-1-1-1-3 network-interface network-1-1-1-1 flow-based n2a-prio-mapping-profile none a2n-prio-mapping-profile none
